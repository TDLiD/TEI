$prefix = "image"
$counter = 1

Get-ChildItem -File | Sort-Object Name | ForEach-Object {
    $ext = $_.Extension
    $newName = "{0}-{1:000}{2}" -f $prefix, $counter, $ext
    Rename-Item $_ $newName
    $counter++
}
